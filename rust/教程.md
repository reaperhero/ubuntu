

# 函数

```
fn main() {
    println!("Hello, world!");
    another_function();
}

fn another_function() {
    println!("Hello, runoob!");
}

fn another_function(x: i32, y: i32) {
    println!("x 的值为 : {}", x);
    println!("y 的值为 : {}", y);
}


```

# mod
```
mod toy {
    pub fn run() {
        println!("run toy");
    }
}

fn main() {
    toy::run();
}
```
